#include <stdio.h>
main()
{
 int A, B;         /* données */
 int X, Y, RESTE;  /* var. d'aide pour l'algorithme d'Euclide */

 do 
    {
      printf("Entrer l'entier A (non nul) : ");
     scanf("%d", &A);
    } 
 while(!A);
 do 
    {
      printf("Entrer l'entier B (non nul) : ");
     scanf("%d", &B);
    } 
 while(!B);
 
 for (RESTE=A, X=A, Y=B ; RESTE ; X=Y, Y=RESTE)
      RESTE = X%Y;
 
 printf("Le PGCD de %d et de %d est %d\n", A, B, X);
 return 0;
}

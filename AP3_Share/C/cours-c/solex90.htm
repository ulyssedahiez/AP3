#include <stdio.h>
main()
{
 /* Déclarations */
 int N;  /* nombre entré */
 char JOUR[8][9] = {"\aErreur!", "lundi", "mardi", "mercredi", 
                    "jeudi", "vendredi", "samedi","dimanche"};
  /* Saisie du nombre */
 printf("Entrez un nombre entre 1 et 7 : ");
 scanf("%d", &N);
 /* Affichage du résultat - pour perfectionnistes */
 if (N>0 && N<8)
     printf("Le %de%c jour de la semaine est %s.\n",
                                   N, (N==1)?'r':' ', JOUR[N]);
 else
     puts(JOUR[0]);
  return 0;
}

Exercice 10.9

#include <stdio.h>
#include <math.h>
 main()
{
 /* Prototypes des fonctions appelées */
 double F(int X);
 /* Variables locales */
 int I;
 /* Traitements */
 printf("\tX\tF(X)\n");
 for (I=1 ; I<=10 ; I++)
      printf("\t%d\t%f\n", I, F(I)); 
 return 0;
}

double F(int X)
{
 return sin(X)+log(X)-sqrt(X);
}

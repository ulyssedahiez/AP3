Exercice 4.4

Solution:

#include <stdio.h>
main()
{
	int A, B, C, AIDE;
	printf("Introduisez trois nombres (A, B, C) : ");
	scanf("%i %i %i", &A, &B, &C);
 	/* Affichage à l'aide de tabulations */
	printf("A = %i\tB = %i\tC = %i\n", A, B, C);
 	AIDE=A;
	A=C;
	C=B;
	B=AIDE;
	printf("A = %i\tB = %i\tC = %i\n", A, B, C);
 	return 0;
}

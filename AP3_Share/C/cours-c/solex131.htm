Affichage: 3 -8 12

3 12 3

Implémentation:

#include <stdio.h>

main()
{
 void MODIFIER(int X, int *Y, int *Z); /* Prototype */ 
  int A,B,C;
 A=3;   
 B=-8;
 C=12; 
 printf("%d %d %d \n", A, B, C);
 MODIFIER(A,&B,&C);
  printf("%d %d %d \n", A, B, C);
 return 0;
}

void MODIFIER(int X, int *Y, int *Z)
{
 int T;
 T  = X;
 X  = *Y;
 *Y = *Z;
 *Z = T;
}

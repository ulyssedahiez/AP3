Exercice 10.13

void LIRE_TAB (int *TAB, int *N, int NMAX)
{
 /* Variables locales */
 int I; 
 /* Saisie de la dimension du tableau */
 do
    {
     printf("Dimension du tableau (max.%d) : ", NMAX);
     scanf("%d", N); /* Attention: écrire N et non &N ! */
    }
 while (*N<0 || *N>NMAX);
 /* Saisie des composantes du tableau */
 for (I=0; I<*N; I++)
    {
     printf("Elément[%d] : ", I);
     scanf("%d", TAB+I);
    }
}

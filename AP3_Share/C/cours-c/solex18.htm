Exercice 4.2

#include <stdio.h>
main()
{
	int JOUR, MOIS, ANNEE, RECU;
	printf("Introduisez la date (JOUR, MOIS, ANNéE) : ");
	RECU=scanf("%i %i %i", &JOUR, &MOIS, &ANNEE);
	printf("\ndonnées reçues : %i\njour  : %i\nmois  : 
	             %i\nannee : %i\n", RECU, JOUR, MOIS, ANNEE);
	return 0;
}
